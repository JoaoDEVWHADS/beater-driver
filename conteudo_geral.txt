Excelente escolha! Criar o arquivo mapa.py vai dar uma vida totalmente nova para o seu audiogame. Para um simulador de Uber por áudio, nós podemos mapear as ruas usando coordenadas geométricas simples (retângulos ou faixas de coordenadas).
Dessa forma, se o carro estiver dentro dos limites de X e Y de uma determinada rua, o jogo sabe o nome dela, o tipo de piso (asfalto ou terra) e qual som deve tocar em loop.
Vamos programar e estruturar o seu arquivo scripts/mapa.py.
1. Criando o arquivo scripts/mapa.py
Copie e cole o código abaixo dentro do seu arquivo mapa.py (que estava com 0 bytes):

Copiar o código
# scripts/mapa.py

class Rua:
    def __init__(self, nome, x_min, x_max, y_min, y_max, tipo_piso="asfalto"):
        self.nome = nome
        self.x_min = x_min
        self.x_max = x_max
        self.y_min = y_min
        self.y_max = y_max
        self.tipo_piso = tipo_piso  # "asfalto" ou "terra"

    def contem_posicao(self, x, y):
        # Verifica se as coordenadas X e Y do carro estão dentro desta rua
        return self.x_min <= x <= self.x_max and self.y_min <= y <= self.y_max


class MapaCidade:
    def __init__(self):
        self.ruas = []
        self.configurar_mapa_padrao()

    def configurar_mapa_padrao(self):
        # Aqui definimos as ruas do jogo usando caixas de colisão/coordenadas.
        # Exemplo: Uma grande avenida de asfalto cruzando com uma rua de terra.
        
        # Avenida Central (Asfalto) - Anda bastante no eixo X
        self.ruas.append(Rua("Avenida Central", -500, 500, -20, 20, "asfalto"))
        
        # Rua dos Buracos (Terra) - Cruza a avenida indo na vertical (eixo Y)
        self.ruas.append(Rua("Rua dos Buracos", -30, 30, -500, 500, "terra"))
        
        # Beco do Chevette (Terra) - Uma rua menor paralela
        self.ruas.append(Rua("Beco do Chevette", 100, 300, 100, 130, "terra"))

    def obter_piso_e_rua(self, x, y):
        # Verifica em qual rua o jogador está com base na posição atual
        for rua in self.ruas:
            if rua.contem_posicao(x, y):
                return rua.tipo_piso, rua.nome
        
        # Se ele sair das ruas mapeadas, podemos dizer que está no "mato" ou terreno baldio (terra)
        return "terra", "Terreno Baldio"

# 1. Verifica onde o carro está no mapa
piso_detectado, nome_da_rua = mapa.obter_piso_e_rua(carro.x, carro.y)

# 2. Se o jogador mudou de rua, avisa usando o leitor de tela/fala
if nome_da_rua != rua_atual_nome:
    rua_atual_nome = nome_da_rua
    audio.falar(f"Entrou na {rua_atual_nome}")

# 3. Controla o volume dos loops baseado no piso e se o carro está andando
if carro.motor_ligado and carro.velocidade > 0.5:
    # Ajusta o volume máximo dependendo da velocidade (mais rápido = som mais alto)
    volume_alvo = min(carro.velocidade / 50.0, 0.8) 
    
    if piso_detectado == "asfalto":
        canal_asfalto.set_volume(volume_alvo)
        canal_terra.set_volume(0.0)  # Zera o da terra
    else:
        canal_terra.set_volume(volume_alvo)
        canal_asfalto.set_volume(0.0) # Zera o do asfalto
else:
    # Se o carro parar ou desligar, os sons de rolagem do pneu ficam mudos
    canal_asfalto.set_volume(0.0)
    canal_terra.set_volume(0.0)

import pygame
import random
import os

class Jogador:
    def __init__(self, audio):
        self.audio = audio
        self.x = 0.0
        self.y = 0.0
        self.velocidade = 2.0
        self.ultimo_passo = 0
        
        # Carrega os efeitos sonoros de passos baseados nas suas pastas
        self.sons_asfalto = []
        self.sons_terra = []
        
        for i in range(1, 4):
            caminho_asfalto = f"audio/jogador/passos_asfalto/passo_asfalto_{i}.wav"
            caminho_terra = f"audio/jogador/passos_terra/passo_terra_{i}.wav"
            if os.path.exists(caminho_asfalto):
                self.sons_asfalto.append(pygame.mixer.Sound(caminho_asfalto))
            if os.path.exists(caminho_terra):
                self.sons_terra.append(pygame.mixer.Sound(caminho_terra))

    def atualizar(self, teclas, mapa):
        tempo_atual = pygame.time.get_ticks()
        movendo = False

        if teclas[pygame.K_w] or teclas[pygame.K_UP]:
            self.y -= self.velocidade
            movendo = True
        if teclas[pygame.K_s] or teclas[pygame.K_DOWN]:
            self.y += self.velocidade
            movendo = True
        if teclas[pygame.K_a] or teclas[pygame.K_LEFT]:
            self.x -= self.velocidade
            movendo = True
        if teclas[pygame.K_d] or teclas[pygame.K_RIGHT]:
            self.x += self.velocidade
            movendo = True

        if movendo and tempo_atual - self.ultimo_passo > 450:
            piso, nome_rua = mapa.obter_piso_e_rua(self.x, self.y)
            
            # Escolhe o som aleatório da pasta correspondente
            if piso == "asfalto" and self.sons_asfalto:
                som = random.choice(self.sons_asfalto)
            elif piso == "terra" and self.sons_terra:
                som = random.choice(self.sons_terra)
            else:
                som = None
                
            if som:
                self.audio.canal_passos.play(som)
                
            self.ultimo_passo = tempo_atual

def gerar_bipe_codigo(frequencia, duracao, vol_esq=1.0, vol_dir=1.0):
    amostragem = 44100
    num_amostras = int(duracao * (amostragem / 1000.0))

    canal_terra.play(som_loop_terra, loops=-1)
    canal_terra.set_volume(0.0)
except Exception as e:
    print(f"Aviso: Não foi possível carregar os loops de ambiente: {e}")

def gerar_bipe_codigo(frequencia, duracao, vol_esq=1.0, vol_dir=1.0):
    amostragem = 44100
    num_amostras = int(duracao * (amostragem / 1000.0))
    t = np.linspace(0, duracao / 1000.0, num_amostras, False)
    onda = np.sin(frequencia * t * 2 * np.pi)
    onda_int = (onda * 16384).astype(np.int16)
    estereo = np.column_stack((onda_int, onda_int))
    estereo[:, 0] = (estereo[:, 0] * vol_esq).astype(np.int16)
    estereo[:, 1] = (estereo[:, 1] * vol_dir).astype(np.int16)
    return pygame.sndarray.make_sound(estereo)

            pygame.quit()
            sys.exit()

    elif estado_jogo == "JOGANDO":
        # Determina quem está se movimentando (Jogador a pé ou Carro)

        piso_detectado, nome_da_rua = mapa.obter_piso_e_rua(pos_x, pos_y)

            rua_atual_nome = nome_da_rua
            audio.falar(f"Entrou na {rua_atual_nome}")

        # Controle de volume dos loops de pneu/pista (apenas se estiver no carro)
        if carro.no_carro and carro.motor_ligado and carro.velocidade > 0.5:
            volume_alvo = min(carro.velocidade / 60.0, 0.7)

                canal_asfalto.set_volume(volume_alvo)
                canal_terra.set_volume(0.0)
            else:
                canal_terra.set_volume(volume_alvo)
                canal_asfalto.set_volume(0.0)
        else:
            canal_asfalto.set_volume(0.0)
            canal_terra.set_volume(0.0)

        # --- LÓGICA DO GPS E CORRIDA ---
        dx = alvo_x - carro.x
        dy = alvo_y - carro.y
        distancia_restante = math.hypot(dx, dy)

        angulo_alvo_deg = math.degrees(math.atan2(dx, -dy)) % 360
        erro_direcao = (angulo_alvo_deg - carro.angulo + 180) % 360 - 180

            else:
                gerar_bipe_codigo(600, 40, 0.4, 0.4).play() # Rota Alinhada

        if carro.no_carro and carro.cortando_giro and tempo_atual % 4000 < 20:
            audio.falar("O cliente reclama do barulho do motor!")

        if carro.no_carro and tempo_atual % 6000 < 20 and distancia_restante > 25:
            audio.falar(f"GPS: Faltam {int(distancia_restante)} metros.")

        if distancia_restante <= 25.0:
            estado_jogo = "CHEGOU"
            carro.velocidade = 0.0
            audio.falar("Você chegou ao destino do passageiro! Corrida finalizada com sucesso. Pressione ENTER para retornar ao menu.")

    elif estado_jogo == "CHEGOU":
        canal_asfalto.set_volume(0.0)
        canal_terra.set_volume(0.0)
        if teclas[pygame.K_RETURN]:
            alvo_x = random.uniform(-200, 200)
            alvo_y = random.uniform(-200, 200)
            estado_jogo = "MENU"
            audio.falar("Menu Principal.")

def gerar_bipe_codigo(frequencia, duracao, vol_esq=1.0, vol_dir=1.0):
    amostragem = 44100
    num_amostras = int(duracao * (amostragem / 1000.0))
    t = np.linspace(0, duracao / 1000.0, num_amostras, False)
    onda = np.sin(frequencia * t * 2 * np.pi)
    onda_int = (onda * 16384).astype(np.int16)
    estereo = np.column_stack((onda_int, onda_int))
    estereo[:, 0] = (estereo[:, 0] * vol_esq).astype(np.int16)
    estereo[:, 1] = (estereo[:, 1] * vol_dir).astype(np.int16)
    return pygame.sndarray.make_sound(estereo)

def gerar_bipe_codigo(frequencia, duracao, vol_esq=1.0, vol_dir=1.0):
    amostragem = 44100
    num_amostras = int(duracao * (amostragem / 1000.0))
    t = np.linspace(0, duracao / 1000.0, num_amostras, False)
    onda = np.sin(frequencia * t * 2 * np.pi)
    onda_int = (onda * 16384).astype(np.int16)
    estereo = np.column_stack((onda_int, onda_int))
    estereo[:, 0] = (estereo[:, 0] * vol_esq).astype(np.int16)
    estereo[:, 1] = (estereo[:, 1] * vol_dir).astype(np.int16)
    return pygame.sndarray.make_sound(estereo)

Achei o culpado! Analisando a física do seu carro.py, deu para entender exatamente por que o Chevette empaca nos 26 km/h e não passa de segunda.
O problema está na forma como a velocidade máxima da primeira marcha e o cálculo do RPM estão conversando. Veja só como estava o cálculo do RPM da roda na linha 173:

rpm_base_roda = lenta_atual + ((self.velocidade / limite_marcha) * (self.rev_limiter - lenta_atual))
O que estava acontecendo?
Na primeira marcha (1), a velocidade máxima configurada é 45 km/h, e o carro sobe de marcha quando o motor atinge 3200 RPM.
Quando você acelera, a velocidade chega perto de 25 km/h ou 26 km/h. Nesse exato momento, o cálculo joga o RPM para 3200.
O carro obedece e passa para a segunda marcha.
Só que na segunda marcha (2), a velocidade máxima configurada salta para 80 km/h e o rpm_down (mínimo para reduzir) está travado em 1800 RPM.
Como o carro acabou de passar para a segunda a 26 km/h, o motor cai para menos de 1800 RPM instantaneamente. A lógica lê isso e força o carro a reduzir para a primeira de novo!
O seu Chevette entrou em um "loop infinito": ele põe segunda a 26 km/h, o giro cai, ele joga a primeira de volta, o giro sobe, ele põe segunda... e o carro não sai do lugar!
A Solução
Ajustei as tabelas de marchas (self.config_marchas_auto) para as velocidades reais de um Chevette automático de 3 marchas (já que os automáticos da época não tinham 5 marchas, usei relações bem escalonadas para as 3 posições principais e limpei as marchas 4 e 5 fantasmas).
Também adicionei uma tolerância de aceleração no cálculo do RPM da roda para que o motor não perca o fôlego assim que a marcha engata.

            else:
                self.x += math.sin(radianos) * self.velocidade * fator_movimento
                self.y -= math.cos(radianos) * self.velocidade * fator_movimento

        if self.pipocos_restantes_tirada > 0:
            if tempo_atual - self.ultimo_pipoco_tirada > random.randint(70, 130):
                self.audio.tocar_pipoco_corte()
                self.pipocos_restantes_tirada -= 1
                self.ultimo_pipoco_tirada = tempo_atual

        if self.motor_ligado:
            rpm_normal = max(0.0, min(1.0, (self.rpm / self.rpm_max) * 1.10))
            self.audio.motor_rpm_velho(rpm_normal, self.cortando_giro)

def gerar_bipe_codigo(frequencia, duracao, vol_esq=1.0, vol_dir=1.0):
    amostragem = 44100
    num_amostras = int(duracao * (amostragem / 1000.0))
    t = np.linspace(0, duracao / 1000.0, num_amostras, False)
    onda = np.sin(frequencia * t * 2 * np.pi)
    onda_int = (onda * 16384).astype(np.int16)
    estereo = np.column_stack((onda_int, onda_int))
    estereo[:, 0] = (estereo[:, 0] * vol_esq).astype(np.int16)
    estereo[:, 1] = (estereo[:, 1] * vol_dir).astype(np.int16)
    return pygame.sndarray.make_sound(estereo)
